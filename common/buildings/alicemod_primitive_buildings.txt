	base_cap_amount = 1
	base_buildtime = 400
	can_demolish = yes
	can_be_ruined = no
	position_priority = 1

	potential = {
		exists = owner
		owner = {
			is_primitive = yes
			has_country_flag = stone_age
		}
	}

	destroy_trigger = {
		exists = owner
		owner = {
			is_primitive = no
		}
	}

	planet_modifier = {
		planet_amenities_add = 5
		planet_housing_add = 4
		planet_crime_add = 1
		planet_max_buildings_add = 1
	}

	upgrades = {
	}
	
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_regular_empire = yes }
		}
		modifier = {
			job_clerk_add = 1
		}
	}
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_gestalt = yes }
		}
		modifier = {
			job_maintenance_drone_add = 1
		}
	}

	resources = {
		category = planet_buildings
		cost = {
			minerals = 200
		}
		upkeep = {
			energy = 1
		}
	}
	
	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_gestalt = yes }
		}
		text = job_maintenance_drone_effect_desc
	}
	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_gestalt = no }
		}
		text = job_clerk_effect_desc
